    //@单链表 C语言版

#ifndef C_SINGLELINK_H
#define C_SINGLELINK_H

#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

//单链表的基本结构
typedef struct Link
{
    int data;//数据域
    struct Link *next;//结点
}link;//新名字

//初始化创建链表
link *createLink()
{
    link *link1 = (link*)malloc(sizeof(link));//分配内存
    if(!link1)//检查内存分配成功情况
    {
        printf("Memory Error!\n");
        return NULL;
    }
    link1->data=0;//初始化
    link1->next=NULL;

    link *p = link1;
    for(int i=0;i<3;i++)//为测试，给链表赋初始值
    {
        link *a = (link*)malloc(sizeof(link));
        a->data = i+1;
        a->next = NULL;

        p->next = a;
        p=p->next;
    }
    return link1;
}

//判断是否为空
bool isEmpty(link *link1)
{
    return (link1==NULL);
}

//插入元素
void insertElem(link *link1,int position,int elem)
{
    if(position<1)//判断位置
    {
        printf("Error Position\n");
    }
    else
    {
        link *p = link1;
        for(int i=0;i<position-1;i++)//遍历到要插入的位置处
        {
            if(p==NULL)
            {
                printf("Wrong Position\n");
                exit(-1);
            }
            p=p->next;
        }
        link *link2 = (link*)malloc(sizeof(link));
        if(!link2)
        {
            printf("Temp Link Memory Error!\n");
        }
        //插入结点
        else{
            link2->data = elem;
            link2->next = p->next;
            p->next = link2;
        }
    }
}

//删除元素
int deleteElem(link *link1,int position)
{
    if(!isEmpty(link1))//判断是否为空
    {
        if(position<1)//判断位置
        {
            printf("Error Position\n");
            return -1;
        }
        else
        {
            link *p = link1;
            for(int i=0;i<position-1;i++)//遍历到要删除的地方
            {
                if(p==NULL)
                {
                    printf("Error Position\n");
                    return -1;
                }
                p=p->next;
            }
            int data = p->next->data;//保存被删除的值
            link *temp = (link*)malloc(sizeof(link));
            temp->next = p->next;
            p->next=p->next->next;//重新链接链表
            free(temp);//释放被删除的结点
            return data;//返回数据
        }
    } else
    {
        printf("Empty Link\n");
        return -1;
    }
}

//查找元素第一次出现的位置
int find_position(link *link1,int elem)
{
    if(!isEmpty(link1))//判断非空
    {
        link *p = link1;
        for(int i=0;p!=NULL;i++)
        {
            p=p->next;
            if(elem == p->data)
            {
                return i+1;
            }

        }
    }
    else
    {
        return -2;
    }
    return 0;
}

//遍历显示全部信息
void display(link *link1)
{
    if(!isEmpty(link1))//判断空链
    {
        link *p = link1;
        while (p->next)
        {
            p=p->next;
            printf("%d ",p->data);
        }
        printf("\n");
    } else
    {
        printf("Empty!\n");
    }
}


#endif //C_SINGLELINK_H

    //@单链表CPP版
// Created by abel on 2019/10/14.

#ifndef C___SINGLELINK_H
#define C___SINGLELINK_H

#include <iostream>
using namespace std;

typedef struct Node
{
    int data;
    Node *next;
}link;

class Link
{
private:
    link *link1;
public:
    //构造函数
    Link();
    //插入元素
    void insertElem(int position,int elem);
    //查看是否为空
    bool isEmpty();
    //得到固定位置的元素
    int getElem(int position);
    //删除元素
    int deleteElem(int position);
    //遍历显示全部元素
    void display();
};

Link::Link() {
    link1 = new link;
    link1->data = 0;
    link1->next = nullptr;
}

void Link::insertElem(int position, int elem) {
    if(position<1)
    {
        cout<<"Error Position"<<endl;
    }
    else
    {
        link *p = link1;
        for(int i=0;i<position-1;i++)
        {
            if(p->next== nullptr)
            {
                cout<<"Error Position!"<<endl;
            }
            p = p->next;
        }
        link *temp = new link;
        temp->data=elem;
        temp->next = p->next;
        p->next=temp;
    }
}

bool Link::isEmpty() {
    return (link1->next== nullptr);
}

int Link::getElem(int position) {
    if(!isEmpty())
    {
        if(position<1)
        {
            cout<<"Error Position!"<<endl;
            return -1;
        }
        else
        {
            link *p = link1;
            for(int i=0;i<position-1;i++)
            {
                if(p->next== nullptr)
                {
                    cout<<"Error Position"<<endl;
                    break;
                }
                else
                {
                    p=p->next;
                }
            }
            p=p->next;
            return p->data;
        }
    }
    else
    {
        cout<<"Empty!"<<endl;
        return -1;
    }
}

int Link::deleteElem(int position){
    if(!isEmpty())
    {
        if(position<1)
        {
            cout<<"Error Position"<<endl;
            exit(-1);
        }
        else
        {
            link *p = link1;
            for(int i=0;i<position-1;i++)
            {
                if(p->next == nullptr)
                {
                    cout<<"Error Position";
                    break;
                }
                else
                {
                    p=p->next;
                }
            }
            link *temp = new link;
            temp->next = p->next;
            temp->data = p->next->data;
            p->next = p->next->next;
            int t = temp->data;
            delete temp;
            return t;
        }
    }
    else
    {
        cout<<"Empty!"<<endl;
        exit(-1);
    }
}

void Link::display() {
    if(!isEmpty())//判断是否为空
    {
        link * p = link1;
        while (p->next)
        {
            p=p->next;
            cout<<p->data<<" ";
        }
        cout<<endl;
    }
    else
    {
        cout<<"Error! Empty!"<<endl;
    }
}
