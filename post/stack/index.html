    顺序栈将以四种语言呈现，但不会有太多注释，毕竟比较简单

    • C语言版本
    //@顺序栈 C语言版
// Created by abel on 2019/10/15.
//

#ifndef C_STACK_H
#define C_STACK_H

#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
//默认的最大长度
#define SIZE 10
//结点
typedef struct Stack
{
    int *data;
    int top;
    int capacity;
}stack;

//创建链表
stack *createStack()
{
    stack *stack1 = (stack*)malloc(sizeof(stack));
    if(!stack1)//检查空间开辟情况
    {
        printf("Memory Error!\n");
        return NULL;
    }
    stack1->capacity = SIZE;//最大长度
    stack1->data = (int*)malloc(sizeof(int)*stack1->capacity);//开辟数组
    stack1->top=-1;//栈位置标志
    if(!stack1->data)
    {
        printf("Memory Error!\n");
        return NULL;
    }
    for(int i=0;i<3;i++)//初始化赋值进去
    {
        stack1->data[i]=i+1;
    }
    stack1->top=2;//加两个
    return stack1;
}

//判断为空
bool isEmpty(stack *stack1)
{
    return (stack1->top==-1);
}

//判断为满
bool isFull(stack *stack1)
{
    return stack1->top==stack1->capacity-1;
}

//入栈
void push(stack * stack1,int elem)
{
    if(!isFull(stack1))
    {
        stack1->data[stack1->top+1]=elem;
        stack1->top++;
    } else
    {
        printf("Full\n");
    }
}
//出栈
int pop(stack* stack1)
{
    if(!isEmpty(stack1))
    {
        return stack1->data[stack1->top--];
    }
    printf("Error\n");
    return -1;
}

//得到栈顶元素
int getElem(stack *stack1)
{
    if(!isEmpty(stack1))
    {
        return stack1->data[stack1->top];
    }
    printf("Error\n");
    return -1;
}

//显示所有数据
void display(stack *stack1)
{
    if(!isEmpty(stack1))
    {
        for(int i=0;i<=stack1->top;i++)
        {
            printf("%d ",stack1->data[i]);
        }
        printf("\n");
    }
    else
    {
        printf("Empty\n");
        return;
    }
}
#endif //C_STACK_H

#ifndef C___STACK_H
#define C___STACK_H

#include <iostream>
using namespace std;

const int SIZE = 10;
class Stack
{
private:
    int *data;
    int capacity;
    int top;
public:
    Stack()//创建栈
    {
        capacity = SIZE;
        data = new int[capacity];
        top=-1;
    }

    bool isEmpty();//栈空

    bool isFull();//栈满

    void push(int elem);//入栈

    int getElem();//得到最上面元素

    int pop();//出栈

    void display();//显示所以元素，做检查
};

bool Stack::isEmpty() {
    return top==-1;
}

bool Stack::isFull() {
    return top+1==capacity;
}

void Stack::push(int elem) {
    if(!isFull())
    {
        data[top+1]=elem;
        top++;
    }
    else
    {
        cout<<"Full!"<<endl;
    }
}

int Stack::getElem() {
    if(!isEmpty())
    {
        return data[top];
    } else
    {
        cout<<"empty"<<endl;
        return -1;
    }
}

int Stack::pop() {
    if(!isEmpty())
    {
        return data[top--];
    }
    else
    {
        cout<<"empty"<<endl;
        return -1;
    }
}

void Stack::display() {
    if(!isEmpty())
    {
        for(int i=0;i<=top;i++)
        {
            cout<<data[i]<<" ";
        }
        cout<<endl;
    }
    else
    {
        cout<<"Empty!\n";
        return;
    }
}

//Created by abel on 2019/10/15.
//
public class Stack
{
	final int SIZE = 10;
	private int top;
	private int capacity;
	private int[] data;
	
	Stack()
	{
		this.capacity = SIZE;
		this.data = new int[capacity];
		this.top = -1;
	}
	
	public boolean isEmpty()
	{
		return (top==-1);
	}
	public boolean isFull()
	{
		return (top==capacity-1);
	}
	public void enlarge()
	{
		this.capacity *= 2;
		int []temp = new int[top+1];
		for(int i=0;i<top;i++)
		{
			temp[i]=this.data[i];
		}
		this.data = new int[this.capacity];
		for(int i=0;i<top;i++)
		{
			this.data[i] = temp[i];
		}
	}
	public void push(int elem)
	{
		if(!isFull())
		{
			data[top+1]=elem;
			top++;
		}
		else
		{
			System.out.println("Enlarge...");
			enlarge();
			data[top+1]=elem;
			top++;
		}
	}
	public int getElem()
	{
		if(!isEmpty())
		{
			return data[top];
		}
		else
		{
			System.out.println("Empty!");
			return -1;
		}
	}
	public int pop()
	{
		if(!isEmpty())
		{
			return data[top--];
		}
		else
		{
			System.out.println("Empty!");
			return -1;
		}
	}
}
