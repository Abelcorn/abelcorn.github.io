#ifndef LinerList_h
#define LinerList_h

#include <iostream>
using namespace std;

const int MAXSIZE = 10;

template <typename T>
class LinerList
{
private:
    T *data;
    int capacity;
    int size;
public:
    LinerList()
    {
        this->data = new T[MAXSIZE];
        this->capacity = MAXSIZE;
        this->size = 0;
    }
    LinerList(int newSize)
    {
        this->capacity = newSize;
        this->data = new T[this->capacity];
        this->size = 0;
    }
    ~LinerList()
    {
        delete []data;
    }
    
    bool isEmpty();
    bool isFull();
    void InsertElem(int position, T elem);
    T DeleteElem(int position);
    int FindElemPosition(T elem);
    int CountElem(T elem);
    void DisplayList();
};


template <typename T>
bool LinerList<T>::isEmpty()
{
    return this->size==0;
}
template <typename T>
bool LinerList<T>::isFull()
{
    return this->size==this->capacity;
}

template <typename T>
void LinerList<T>::InsertElem(int position, T elem)
{
    if(position<1||position>size+1)
    {
        cout<<"Postion Error!\n";
        return;
    }
    if(isFull())
    {
        cout<<"List is Full!\n";
        return;
    }
    for(int i=size;i>=position-1;i--)
    {
        data[i+1] = data[i];
    }
    data[position-1] = elem;
    size++;
}

template <typename T>
T LinerList<T>::DeleteElem(int position)
{
    if(position<1||position>size+1)
    {
        cout<<"Position Error!\n";
        return -1;
    }
    if(isEmpty())
    {
        cout<<"List is Empty!\n";
        return -1;
    }
    for(int i=position-1;i<size;i++)
    {
        data[i]=data[i+1];
    }
    size--;
}
template <typename T>
int LinerList<T>::FindElemPosition(T elem)
{
    if(isEmpty())
    {
        cout<<"Empty List!\n";
        return -1;
    }
    for(int i=0;i<size;i++)
    {
        if(data[i]==elem)return (i+1);
    }
    cout<<"No The elem\n";
    return -1;
}

template <typename T>
int LinerList<T>::CountElem(T elem)
{
    if(isEmpty())
    {
        cout<<"Empty List!\n";
        return -1;
    }
    int j = 0;
    for(int i=0;i<size;i++)
    {
        if(data[i] == elem)j++;
    }
    if(j==0)
    {
        cout<<"No element!\n";
        return -1;
    }
    return j;
}

template <typename T>
void LinerList<T>::DisplayList()
{
    if(!isEmpty())
    {
        for(int i=0;i<size;i++)
        {
            cout<<data[i]<<" ";
        }
        cout<<endl;
        return;
    }
    cout<<"Empty List!\n";
}
