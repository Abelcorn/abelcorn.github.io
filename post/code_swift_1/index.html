Swift初识

作为程序员，学习语言免不了学习"Hello World!"，而swift的这个和Python一样简单，只要一句：

print("Hello World!")

print是自带换行的。

简单值

变量的声明:

  • 为了使用常量，我们使用let来进行声明，类似的，变量使用var来声明：
let myConstant = 42
// 常量无法再次被赋值改变了
var myVariable = 42
myVariable = 45 // 而变量可以随意改变
  • 该语言是类型自推断的，如果你习惯于C/C++语言的书写，那么你也可以指定类型:
let aConstant: Double = 70.0  //在名字后面冒号再加类型
  • 但是需要注意的是，一个值不会被隐式的转换为其他类型，也就是需要你强制转换，如：
let label = "The width is "
let width = 96
let widthLabel = label + String(width) //如果你尝试不加String()进行转换，那么将会报错
print(widthLabel)
  • 这里有一种更有效的字符串转换方法
let apples = 5
let oranges = 4
let appleSum = "I have \(apples) apples"  //通过\()的方式在字符串中插入信息
let fruitSum = "I have \(apples+oranges) pieces of fruit."
print(appleSum)
print(fruitSum)

函数和闭包

  • 使用func来声明函数，使用->来指定返回值的类型
func greet(name: String,day: String)->String //表示函数类型为String
{
    return "Hello \(name),today is \(day)."
}
print(greet(name: "Bob", day: "Tuesday"))

//多个返回值
func calculateStatistics(scores:[Int])->(min:Int,max:Int,sum:Int)
{
    var min = scores[0]
    var max = scores[0]
    var sum = 0
    for score in scores
    {
        if score>max{
            max = score
        }else if (score<min){
            min = score
        }
        sum += score
    }
    return (min,max,sum)
}

let numbers = calculateStatistics(scores: [5,3,100,3,9])
print(numbers.sum)//可以显式调用此数据
print(numbers.0)//或者用下标调用，等价于number.min

//多个参数
func sumOf(numbers:Int...)->Int//可变个数参数
{
    var sum = 0
    for number in numbers
    {
        sum += number
    }
    return sum
}
print(sumOf()) //输出为0
print(sumOf(numbers:42,597,12)) //输出为651

  • 类的继承
//这是父类
class NamedShape
{
    var numberOfSides:Int = 0
    var name: String
    
    init(name:String)//构造函数，使用init()函数
    {
        self.name = name; //self就类似于this指针
    }
    func desc() -> String
    {
        return "A shape with \(numberOfSides) sides."
    }
    deinit { //析构函数
        self.numberOfSides = 0
    }
}
//继承
class Square: NamedShape//通过冒号继承
{
    var sideLength: Double
    
    init(sideLength: Double, name: String)
    {
        self.sideLength = sideLength
        super.init(name: name)
        numberOfSides = 4
    }
    func area() -> Double
    {
        return sideLength * sideLength
    }
    override func desc() -> String //如果需要改写父类的方法，需要添加override关键字，否则会报错
    {
        return "A square with sides of length \(sideLength)."
    }
}
let test = Square(sideLength: 5.2, name: "My test square")
print(Int(test.area()))
print(test.desc())

枚举和结构体

枚举类型 enum

  • 枚举类型可以定义方法
enum Rank: Int
{
    case Ace = 1 //第一个出现的case就是rawValue
    case Two, Three, Four, Five, Six, Seven, Eight, Nine, Ten
    case Jack, Queen, King
    func Desc()->String
    {
        switch self {
        case .Ace:
            return "ace"
        case .Jack:
            return "jack"
        case .Queen:
            return "queen"
        case .King:
            return "king"
        default:
            return String(self.rawValue)
        }
    }
}
let ace = Rank.Ace
let aceRawValue = ace.rawValue
print(ace)
print(aceRawValue)

协议和扩展

使用protocol来声明一个协议

protocol ExampleProtocol{
    var simple: String{get}
    mutating func adjust()
}
//类、枚举都可以实现协议
class SimpleClass: ExampleProtocol
{
    var simple: String = "A very simple class"
    var anotherProperty: Int = 69105
    func adjust() {
        simple += " Now 100% adjusted."
    }
}
var a = SimpleClass()
a.adjust()
let aDesc = a.simple

struct SimpleStructure: ExampleProtocol{
    var simple: String = "A simple structure"
    mutating func adjust() {
        simple += " (adjusted)"
    }
}

var b = SimpleStructure()
b.adjust()
let bDesc = b.simple

使用扩展

extension Int: ExampleProtocol{
    var simple: String{
        return "The number \(self)"
    }
    mutating func adjust() {
        self += 42
    }
}
print(7.simple)

泛型

在尖括号中创建泛型

//尖括号表示类型的代替名
func repeatItem<Item>(item:Item, numberOfTimes: Int) -> [Item]{//方括号表示返回值是泛型类型
    var result = [Item]()
    for _ in 0..<numberOfTimes{
        result.append(item)
    }
    return result
}
repeatItem(item: "Knock", numberOfTimes: 4)
