    #include <stdio.h>
#include <stdlib.h>

//创建一个顺序表结构
typedef struct Table
{
    int *data;//存数据
    int length;//表长度
}table;

//冒泡排序
void Bubble_Sort(int a[],int n)
{
    int temp;
    int i,j;
    printf("Original: ");
    for(i=0;i<n;i++)
    {
        printf("%d ",a[i]);
    }
    printf("\n");
    //先将原始数据打印出来
    //排序
    for(i=0;i<n;i++)
    {
        for(j=0;j<n-i-1;j++)
        {
            if(a[j]>a[j+1])
            {
                temp = a[j];
                a[j] = a[j+1];
                a[j+1]=temp;
            }
        }
    }
}

//创建顺序表
table *createTable()
{
    table *t = (table*)malloc(sizeof(table));
    if(!t)//判断空间开辟情况
    {
        printf("Memory Error!\n");
        return NULL;
    }
    //输入存入的数据数量
    printf("How many numbers you want to enter?\n");
    scanf("%d",&t->length);
    t->data = (int*)malloc(sizeof(int)*t->length);
    if(!t->data)
    {
        printf("Memory Error!\n");
        return NULL;
    }
    //判断空间开辟情况
    printf("Enter them:\n");
    //输入数据
    for(int i=0;i<t->length;i++)
    {
        scanf("%d",&t->data[i]);
    }
    //调用排序算法
    Bubble_Sort(t->data,t->length);
    printf("Now:    ");
    for(int i=0;i<t->length;i++)
    {
        printf("%d ",t->data[i]);
    }
    printf("\n");
    //展示现在的顺序
    return t;
}

//二分查找
int Binary_Search(const int a[],int x,int n)
{
    int low=0,high=n,mid;
    while(low<=high)//当低位不高于高位时
    {
        mid = (high+low)/2;//求出中间位
        if(x == a[mid])//判断是否是此数
        {
            return mid;//是则返回
        }
        else//不是
        {
            if(x>a[mid])low = mid+1;
            //若要找的数大于中间数，则缩小空间至原来的[mid,high]
            else high = mid-1;
            //若要找的数小于中间数，则缩小空间至原来的[low,mid]
        }
    }
    return -1;//找不到
}

int main()
{
    table *t = createTable();//创建此顺序表
    int x;
    printf("Enter the number which you wanna find: ");
    scanf("%d",&x);//输入你要找的数据
    int flag = Binary_Search(t->data,x,t->length);//存下数据做处理
    if(flag!=-1)//找到了则
    {
        printf("Successful!\n%d 's position is %dth\n",x,flag+1);
    }
    else//没找到
    {
        printf("Failed!\n");
    }
    return 0;
}
